'%bankai%' <- function(x, y) {
  x ^ 2 + y ^ 2
}
transmute(
  periodic_table,
  atomic_number = atomic_number,
  symbol = symbol,
  name = name,
  weight = atomic_weight,
  density = density,
  heat_cap = heat_capacity,
  log = log(heat_cap),   # Logarithms Operation
  integer_vol = weight %/% density,
  remainder_vol = weight %% density,
  vol = weight / density,
  bankai = vol %bankai% heat_cap  # private operator
)

The Rank of an element is the number of elements in the sequence that are smaller than the estimated element.

dplyr::min_rank(c(1, 2, 2, NA, 3, 4))
## [1]  1  2  2 NA  4  5
dplyr::desc(c(1, 3, 7, NA, 5, 3))
## [1] -1 -3 -7 NA -5 -3
dplyr::min_rank(desc(c(1, 2, 2, NA, 3, 4)))
## [1]  5  3  3 NA  2  1
dplyr::dense_rank(c(1, 2, 2, NA, 3, 4))  # No gaps for the NA element
## [1]  1  2  2 NA  3  4
dplyr::row_number(c(1, 2, 2, NA, 3, 4))
## [1]  1  2  3 NA  4  5
dplyr::percent_rank(c(1, 2, 2, NA, 3, 4)) # Convert the rank to proportion
## [1] 0.00 0.25 0.25   NA 0.75 1.00
dplyr::ntile(c(1, 2, 2, NA, 3, 4), 2)
## [1]  1  1  1 NA  2  2
range = c(1, 2, 2, NA, 3, 4)
(test_table <- mutate(
  data.frame(range = c(1, 2, 2, NA, 3, 4)),
  cume_dist = cume_dist(range),
  rownum = row_number(range),
  total_count = sum(!is.na(range)),
  cume_dist_mimic = rank(range, ties.method = "max") / total_count
))
ggplot(test_table) +
  geom_bar(
    mapping = aes(x = rownum, y = cume_dist), 
    stat = "identity"
  )

Alphabet Arrays

tibble(
  y = utf8ToInt("CMYDOXEQWFSV") - 64,
  c = c(1,   2,  3,  1,  2,  3,  1,  2,  3,  1,  2,  3),
  g = c(1,   1,  1,  1,  1,  1,  1,  2,  1,  1,  1,  1)
) %>% mutate(rownum = row_number()) %>%
ggplot(aes(x = rownum, y = y, group = factor(c), color = factor(g))) +
  geom_point() + geom_line()
